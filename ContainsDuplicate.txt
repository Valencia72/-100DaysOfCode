class Solution {
    public boolean containsDuplicate(int[] nums) {
        HashMap<Integer,Integer> hm=new HashMap<>();
        for(int i=0;i<nums.length;i++){
            if(!hm.containsKey(nums[i]))
                hm.put(nums[i],1);
            else{
                hm.put(nums[i],hm.getOrDefault(nums[i],1)+1);
            }
        }
        for(int x:hm.values()){
            if(x>=2){
                return true;
            }
        }
        return false;
    }
}
